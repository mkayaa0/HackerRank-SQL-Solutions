SELECT
    CASE
    WHEN UPPER(OCCUPATION) = 'DOCTOR' THEN CONCAT(NAME, '(D)')
    WHEN UPPER(OCCUPATION) = 'ACTOR' THEN CONCAT(NAME, '(A)')
    WHEN UPPER(OCCUPATION) = 'PROFESSOR' THEN CONCAT(NAME, '(P)')
    WHEN UPPER(OCCUPATION) = 'SINGER' THEN CONCAT(NAME, '(S)')
    END
FROM OCCUPATIONS ORDER BY NAME, OCCUPATION;
SELECT 
    'There are a total of ' || COUNT(*) || ' ' || LOWER(OCCUPATION) || 's.'
FROM OCCUPATIONS
GROUP BY OCCUPATION
ORDER BY COUNT(*), OCCUPATION;
