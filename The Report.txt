(
SELECT S.NAME, G.GRADE, S.MARKS 
FROM STUDENTS S
JOIN GRADES G
	ON S.MARKS BETWEEN G.MIN_MARK AND G.MAX_MARK
WHERE G.GRADE > 7
)
UNION 
(
SELECT NULL, G.GRADE, S.MARKS 
FROM STUDENTS S
JOIN GRADES G
	ON S.MARKS BETWEEN G.MIN_MARK AND G.MAX_MARK
WHERE G.GRADE < 8
ORDER BY GRADE DESC, NAME ASC
);
